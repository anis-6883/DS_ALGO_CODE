#include<iostream>
using namespace std;
int main()
{
    string str;
    getline(cin, str);
    int f = 0, l = str.size()-1;
    string str2 = str;

    while (f < l)
    {
        char temp = str2[f];
        str2[f] = str2[l];
        str2[l] = temp;
        f++;
        l--;
    }

    str == str2 ? cout<<"Palindrome!"<<endl : cout<<"Not a Palindrome!"<<endl;
    return 0;
}
