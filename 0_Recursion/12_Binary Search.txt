#include <iostream>
using namespace std;

bool binarySearch(int arr[], int n, int k, int l, int h){
    if(l <= h){
        int mid = (l + h) / 2;
        if(arr[mid] == k) return true;
        if(arr[mid] < k)
            return binarySearch(arr, n, k, mid+1, h);
        else
            return binarySearch(arr, n, k, l, mid-1);
    }
    return false;
}

int main()
{
    int arr[] = {1, 4, 7, 9, 16, 56, 70};
    int n = sizeof(arr)/sizeof(arr[0]);
    int k = 70;
    binarySearch(arr, n, k, 0, n-1) ? 
    cout << "True" << endl : 
    cout << "Fasle" << endl;
    return 0;
}
