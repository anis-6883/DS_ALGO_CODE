#include <bits/stdc++.h>
using namespace std;

void marge_array(int arr1[],int arr2[], int n, int m)
{
    for(int i=0; i<n; i++)
    {
        bool flag=0;
        if(arr1[i]>arr2[0])
        {
            swap(arr1[i],arr2[0]);
            flag=1;
        }
        if(flag)
        {
            sort(arr2,arr2+m);
        }
    }
}

int main()
{
    int n,m;
    cin>>n>>m;
    int arr1[n];
    int arr2[m];
    for(int i=0; i<n; i++)
        cin>>arr1[i];

    for(int i=0; i<m; i++)
        cin>>arr2[i];

    marge_array(arr1,arr2,n,m);

    for(int i=0; i<n; i++)
        cout<<arr1[i]<<" ";

    for(int i=0; i<m; i++)
        cout<<arr2[i]<<" ";

    return 0;
}
