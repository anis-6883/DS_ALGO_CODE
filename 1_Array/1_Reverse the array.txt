#include<bits/stdc++.h>
using namespace std;
void reverse_arr(int [], int, int);
int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0; i<n; i++)
    {
        cin>>arr[i];
    }
    reverse_arr(arr,0,n-1);
    for(int i=0; i<n; i++)
    {
        cout<<arr[i]<<endl;;
    }
    return 0;
}
void reverse_arr(int arr[], int f, int l)
{
    while (f < l)
    {
        int temp = arr[f];
        arr[f] = arr[l];
        arr[l] = temp;
        f++;
        l--;
    }
}
