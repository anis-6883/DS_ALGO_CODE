// Bobble Sort
// Swap element if wrong order L>R


#include <iostream>
using namespace std;

int main(){

    int arr[] = {12, 45, 23, 51, 19, 8};
    int n = sizeof(arr)/sizeof(arr[0]);
    
    for(int i=0; i<n-1; i++){
        for(int j=0; j<n-i-1; j++){
            if(arr[j]>arr[j+1])
                swap(arr[j], arr[j+1]);
        }
    }

    for(int x: arr) cout<< x <<" ";
    cout<<endl;
    return 0;
}
